% --- DEFINIZIONE MACRO PER LA PAGINA ---
% Creiamo un comando riutilizzabile per l'impaginazione
\newcommand{\UniboTitlePage}[1]{
	\begin{titlepage}
		% Margini simmetrici per la copertina
		\newgeometry{margin=3cm, top=3cm, bottom=3cm}
		
		\begin{center}
			
			% Logo Unibo (L'argomento #1 deciderà quale immagine caricare)
			\includegraphics[width=6cm]{Figures/Theme/#1} \par
			\vspace{1cm}
			
			% Intestazione Ateneo
			{\large \textbf{\GetDepartment} \par}
			\vspace{0.5cm}
			{\Large \textbf{\GetDegree} \par}
			
			\vspace{2.5cm}
			
			% Titolo della Tesi
			{\LARGE \textbf{\GetTitle} \par}
			
			% Sottotitolo (se presente)
			\ifthenelse{\equal{\GetSubtitle}{}}{}{
				\vspace{0.5cm}
				{\Large \textit{\GetSubtitle} \par}
			}
			
			\vspace{1.5cm}
			{\large \textbf{\GetThesisType} \par}
			
			\vspace{2.5cm}
			
			% Relatore e Candidato
			\begin{minipage}[t]{0.45\textwidth}
				\begin{flushleft}
					\large
					\textbf{Relatore:} \\
					\GetSupervisor \\
					\vspace{0.5cm}
					% Controllo se esiste il correlatore prima di stamparlo
					\ifthenelse{\isundefined{\GetCoSupervisor}}{}{
						\textbf{Correlatore:} \\
						\GetCoSupervisor
					}
				\end{flushleft}
			\end{minipage}
			\hfill
			\begin{minipage}[t]{0.45\textwidth}
				\begin{flushright}
					\large
					\textbf{Presentata da:} \\
					\GetFirstAuthor \\
				\end{flushright}
			\end{minipage}
			
			\vfill
			
			% Sessione e Anno Accademico
			\rule{0.6\textwidth}{0.4pt} \\
			\vspace{0.2cm}
			% Usiamo la variabile Date per la "Sessione mese anno" (es. Sessione Marzo 2026)
			{\large \GetDate \par}
			{\large Anno Accademico \GetAcademicYear \par}
			
		\end{center}
		
		\restoregeometry
	\end{titlepage}
}
